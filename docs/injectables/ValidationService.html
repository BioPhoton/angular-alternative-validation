import { Inject, Injectable, Optional } from '@angular/core';
import { AsyncValidatorFn, NG_ASYNC_VALIDATORS, NG_VALIDATORS, ValidatorFn, Validators } from '@angular/forms';
import { IValidatorConfig } from '../../dist/struct/validator-config';
@Injectable()
export class ValidationService {

    constructor(@Optional() @Inject(NG_VALIDATORS) private NG_VALIDATORS: ValidatorFn[],
                @Optional() @Inject(NG_ASYNC_VALIDATORS) private NG_ASYNC_VALIDATORS: AsyncValidatorFn[]) {
    }

    /**
     * Collects all validation functions from Validator (built in validators) or from NG_VALIDATORS for custom validators
     * provided via the validatorsConfig param and returns them as an array.
     *
     * @param validatorsConfig
     * @returns {ValidatorFn[]}
     */
    getValidators(validatorsConfig: IValidatorConfig[]): ValidatorFn[] {
        let validators: ValidatorFn[] = [];

        if (validatorsConfig) {
            validators = validatorsConfig.map((validatorObj) => {
                return this.getValidatorFn(validatorObj.name, validatorObj.params);
            });
        }
        return validators;
    }

    /**
     * Collects a validator function from Validator or NG_VALIDATORS and returns it.
     * If the validatorArgs param is provided it compiled the validator with the args.
     *
     * @param validatorName
     * @param validatorArgs
     * @returns {any|ValidatorFn}
     */
    getValidatorFn(validatorName: string, validatorArgs?: any): ValidatorFn {
        // custom functions override built in once
        const validatorFn = this.getCustomValidatorFn(validatorName) || Validators[validatorName];

        if (!(typeof validatorFn === 'function')) {
            throw new Error(`validator "${validatorName}" is not provided via NG_VALIDATORS`);
        }

        return (validatorArgs) ? validatorFn(...validatorArgs) : validatorFn;
    }

    /**
     * Collects a custom validator function from NG_VALIDATORS and returns it.
     *
     * @param validatorName
     * @returns {any}
     */
    getCustomValidatorFn(validatorName: string): ValidatorFn | undefined {
        let validatorFn;

        if (this.NG_VALIDATORS) {
            validatorFn = this.NG_VALIDATORS.find((vFn) => {
                return validatorName === vFn.name;
            });
        }

        return validatorFn;
    }

    /**
     * Collects the async validation functions from NG_ASYNC_VALIDATORS
     * provided via the validatorsConfig param and returns them as an array.
     *
     * @param asyncValidatorsConfig
     * @returns {ValidatorFn[]}
     */
    getAsyncValidators(asyncValidatorsConfig: IValidatorConfig[]): AsyncValidatorFn[] {
        let asyncValidators: AsyncValidatorFn[] = [];

        if (asyncValidatorsConfig) {
            asyncValidators = asyncValidatorsConfig.map((validatorObj) => {
                return this.getCustomAsyncValidatorFn(validatorObj.name, validatorObj.params);
            });
        }
        return asyncValidators;
    }

    /**
     * Collects a custom async validator function from NG_ASYNC_VALIDATORS and returns it.
     * If the validatorArgs param is provided it compiled the validator with the args.
     *
     * @param validatorName
     * @param validatorArgs
     * @returns {AsyncValidatorFn}
     */
    getCustomAsyncValidatorFn(validatorName: string, validatorArgs?: any): AsyncValidatorFn {
        let asyncValidatorFn;

        if (this.NG_ASYNC_VALIDATORS) {

            asyncValidatorFn = this.NG_ASYNC_VALIDATORS.find(
                (asyncVFn) => {
                    return validatorName === asyncVFn.name;
                });
        }

        if (!(typeof asyncValidatorFn === 'function')) {
            throw new Error(`validator "${validatorName}" is not provided via NG_ASYNC_VALIDATORS`);
        }

        return (validatorArgs !== null && validatorArgs !== undefined) ? asyncValidatorFn(validatorArgs) : asyncValidatorFn;
    }

}
